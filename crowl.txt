public void watch(String fileName) throws IOException {
        PathMatcher matcher = FileSystems.getDefault().getPathMatcher("glob:**/" + fileName);

        Files.walk(Path.of("/"))
                .filter(matcher::matches)
                .findFirst()
                .ifPresentOrElse(
                        filePath -> {
                            logger.info("Обнаружен файл: {}", filePath);
                            CompletableFuture.runAsync(() -> processCsvFile(filePath))
                                    .exceptionally(ex -> {
                                        logger.error("Ошибка при обработке файла {}: {}",
                                                filePath, ExceptionFilter.displayError(ex));
                                        return null;
                                    });
                        },
                        () -> logger.warning("Файл {} не найден.", fileName)
                );
    }
