Highcharts.chart('container', {
    chart: {
        type: 'pie'
    },
    title: {
        text: 'Пример графика с легендой'
    },
    legend: {
        layout: 'horizontal',        // Горизонтальная легенда
        align: 'center',             // Центрирование легенды
        verticalAlign: 'bottom',     // Размещение внизу
        itemStyle: {
            fontWeight: 'bold'
        },
        maxHeight: 150,              // Ограничение высоты легенды
        itemDistance: 10,            // Расстояние между элементами
        symbolWidth: 20,             // Размер символов в легенде
        symbolHeight: 20,            // Высота символов
        labelFormatter: function() {
            return this.name;       // Форматирование текста в легенде
        },
        // Разбиение легенды на 2 столбца (максимально два элемента в каждом)
        itemMarginTop: 5,
        useHTML: true,               // Разрешает использование HTML
        width: 300,                  // Ширина легенды
        reversed: false,
        navigation: {
            activeColor: '#3E576F',  // Цвет активных элементов
            inactiveColor: '#CCC'    // Цвет неактивных элементов
        }
    },
    series: [{
        data: [
            { name: 'Категория 1', y: 25 },
            { name: 'Категория 2', y: 15 },
            { name: 'Категория 3', y: 20 },
            { name: 'Категория 4', y: 30 },
            { name: 'Категория 5', y: 10 }
        ]
    }]
});
