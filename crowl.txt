#!/bin/bash

# Сохраняем рабочую директорию
WORKING_DIR=$(pwd)

# Спрашиваем у пользователя, хотите ли вы выполнить gradle clean build
echo "Хотите выполнить gradle clean build? (y/n)"
read -r RESPONSE

if [ "$RESPONSE" = "y" ]; then
    # Выполняем сборку проекта
    cd "$WORKING_DIR"
    gradle clean build
fi

# Переходим в директорию с скомпилированным jar
cd "$WORKING_DIR/build/libs"

# Запускаем JAR файл
java -jar F*

# Возвращаемся в исходную директорию
cd "$WORKING_DIR"

#!/bin/bash

# Находим PID процесса, связанного с вашим приложением
APP_NAME="myapp.jar"

# Ищем процесс по имени приложения и завершаем его
PID=$(ps aux | grep "$APP_NAME" | grep -v grep | awk '{print $2}')

if [ -n "$PID" ]; then
    echo "Останавливаю приложение с PID $PID..."
    kill -9 "$PID"
else
    echo "Процесс не найден."
fi
