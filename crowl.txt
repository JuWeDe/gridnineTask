private boolean checkTableExists(String tableName) throws SQLException {
    // Проверяем сначала в базе OIS
    try (Connection oisConnection = getConnection("ois");
         Statement statement = oisConnection.createStatement()) {
        ResultSet resultSet = statement.executeQuery("SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = '" + tableName + "'");
        if (resultSet.next()) {
            logger.info("Таблица {} найдена в базе OIS", tableName);
            return true;
        }
    } catch (SQLException e) {
        logger.warn("Таблица {} не найдена в базе OIS", tableName);
    }

    // Если таблица не найдена в OIS, проверяем в базе Core
    try (Connection coreConnection = getConnection("core");
         Statement statement = coreConnection.createStatement()) {
        ResultSet resultSet = statement.executeQuery("SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = '" + tableName + "'");
        if (resultSet.next()) {
            logger.info("Таблица {} найдена в базе Core", tableName);
            return true;
        }
    } catch (SQLException e) {
        logger.warn("Таблица {} не найдена в базе Core", tableName);
    }

    logger.info("Таблица {} не найдена ни в базе OIS, ни в базе Core", tableName);
    return false; // Таблица не найдена ни в одной из баз
}
