import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.GregorianCalendar;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class FileCopierWithCustomPrefixAndFullDateChange {
    
    private static final Pattern FILE_NAME_PATTERN = Pattern.compile("(.*?)_(.*)\\..*");
    private static final SimpleDateFormat DATE_TIME_FORMATTER = new SimpleDateFormat("yyyyMMddHHmm");
    
    public static void main(String[] args) throws IOException {
        String sourceFilePath = "/home/dow/downloads/my_prefix_202012091158.txt"; // Путь к исходному файлу
        File sourceFile = new File(sourceFilePath);
        
        int numberOfCopies = 10; // Количество копий, которое нужно создать
        
        Matcher matcher = FILE_NAME_PATTERN.matcher(sourceFile.getName());
        if (matcher.find()) {
            String customPrefix = matcher.group(1);
            String originalBaseName = matcher.group(2);
            String extension = sourceFile.getName().substring(matcher.end() - 1);
            
            for (int i = 0; i < numberOfCopies; i++) {
                Calendar calendar = new GregorianCalendar();
                calendar.add(Calendar.MINUTE, i); // Меняем временную метку на i минут вперед
                String newDateTime = DATE_TIME_FORMATTER.format(calendar.getTime());
                String newFileName = customPrefix + "_" + newDateTime + "." + extension;
                File targetFile = new File("/home/dow/downloads/FileStorage", newFileName);
                
                Files.copy(sourceFile.toPath(), targetFile.toPath());
                System.out.println("Файл успешно создан: " + targetFile.getAbsolutePath());
            }
        } else {
            System.out.println("Неверный формат имени файла.");
        }
    }
}


### Изменения:
1. Регулярное выражение:
   - Обновлено регулярное выражение (.*?)_(.*)\..*. Оно теперь захватывает любой префикс перед первым символом _ и оставляет его в первой группе захвата.

2. Использование произвольного префикса:
   - Вместо жёстко заданного префикса alal теперь используется любая последовательность символов, найденная в начале имени файла перед первым символом _.

Теперь, независимо от того, какой префикс у исходного файла, программа корректно обработает его и создаст копии с изменённым временем, сохраняя оригинальный префикс.
